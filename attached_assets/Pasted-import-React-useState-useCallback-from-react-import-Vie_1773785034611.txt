const { width } = Dimensions.get('window');
const blurhash = 'L6PZfSi_.AyE_3t7t7R**0o#DgR4';

function FeaturedGem({ product }: { product: Product }) {
  const router = useRouter();
  const { isSaved, toggleSave } = useSaved();
  const saved = isSaved(product.id);

  return (
    <TouchableOpacity
      style={styles.featured}
      activeOpacity={0.96}
      onPress={() => router.push(`/product/${product.id}`)}
    >
      <Image
        source={{ uri: product.images[0] }}
        style={styles.featuredImage}
        contentFit="cover"
        placeholder={blurhash}
        transition={400}
      />
      <LinearGradient
        colors={['rgba(15,15,13,0)', 'rgba(15,15,13,0.7)', 'rgba(15,15,13,0.95)']}
        style={styles.featuredGradient}
      />
      <View style={styles.featuredContent}>
        <View style={styles.gemLabel}>
          <Ionicons name="diamond" size={10} color={Colors.gemsAccent} />
          <Text style={styles.gemLabelText}>GEM OF THE DAY</Text>
        </View>
        <Text style={styles.featuredTitle}>{product.title}</Text>
        {product.shop && (
          <Text style={styles.featuredShop}>{product.shop.name} · {product.shop.city}</Text>
        )}
        <Text style={styles.featuredPrice}>
          {formatPrice(product.price, product.currency)}
        </Text>
      </View>
      <TouchableOpacity
        style={[styles.featuredSave, saved && styles.featuredSaveActive]}
        onPress={(e) => { e.stopPropagation(); toggleSave(product); }}
      >
        <Ionicons
          name={saved ? 'heart' : 'heart-outline'}
          size={20}
          color={saved ? Colors.error : '#fff'}
        />
      </TouchableOpacity>
    </TouchableOpacity>
  );
}

export default function HiddenGems() {
  const [category, setCategory] = useState<'all' | ProductCategory>('all');
  const [refreshing, setRefreshing] = useState(false);
  const [products, setProducts] = useState<Product[]>(() => getHiddenGemsFeed());

  const featured = products[0];
  const rest = products.slice(1);

  const onRefresh = useCallback(() => {
    setRefreshing(true);
    setTimeout(() => {
      setProducts(getHiddenGemsFeed(category === 'all' ? undefined : category));
      setRefreshing(false);
    }, 700);
  }, [category]);

  function handleCategoryChange(cat: 'all' | ProductCategory) {
    setCategory(cat);
    setProducts(getHiddenGemsFeed(cat === 'all' ? undefined : cat));
  }

  // Pair items for 2-column grid
  const pairs: [Product, Product | null][] = [];
  for (let i = 0; i < rest.length; i += 2) {
    pairs.push([rest[i], rest[i + 1] ?? null]);
  }

  const renderHeader = () => (
    <View>
      {/* Dark header */}
      <View style={styles.header}>
        <View>
          <Text style={styles.headerTitle}>Hidden Gems</Text>
          <Text style={styles.headerSub}>Rare finds. Small shops. Big stories.</Text>
        </View>
        <View style={styles.headerIcon}>
          <Ionicons name="diamond" size={22} color={Colors.gemsAccent} />
        </View>
      </View>

      <CategoryFilter selected={category} onSelect={handleCategoryChange} dark />

      {featured && <FeaturedGem product={featured} />}

      <View style={styles.sectionHeader}>
        <Text style={styles.sectionTitle}>More Gems</Text>
        <Text style={styles.sectionCount}>{rest.length} pieces</Text>
      </View>
    </View>
  );

  return (
    <View style={styles.root}>
      <SafeAreaView style={styles.safeTop}>
        <View style={styles.topBar}>
          <Text style={styles.wordmark}>KIKI</Text>
          <Ionicons name="diamond-outline" size={22} color={Colors.gemsAccent} />
        </View>
      </SafeAreaView>

      <FlatList
        data={pairs}
        keyExtractor={(_, index) => index.toString()}
        renderItem={({ item: [left, right] }) => (
          <View style={styles.row}>
            <ProductCardSmall product={left} />
            {right ? (
              <ProductCardSmall product={right} />
            ) : (
              <View style={{ flex: 1 }} />
            )}
          </View>
        )}
        ListHeaderComponent={renderHeader}
        showsVerticalScrollIndicator={false}
        refreshControl={
          <RefreshControl
            refreshing={refreshing}
            onRefresh={onRefresh}
            tintColor={Colors.gemsAccent}
          />
        }
        ItemSeparatorComponent={() => <View style={{ height: 3 }} />}
        contentContainerStyle={styles.listContent}
      />
    </View>
  );
}
